Model Associations

1. Patient
	has_and_belongs_to_many :appointments
	has_and_belongs_to_many :doctors
	has_many :records
	has_many :treatments
	belongs_to :medical_center #foreign key

2. Appointment
	has_one :patient
	has_one :doctor
	has_many :records
	has_many :treatments
	belongs_to :medical_center #foreign key

3. Doctor
	has_many :patients
	has_many :appointments
	has_many :records
	has_many :treatments
	belongs_to :medical_center #foreign key

4. Records
	has_and_belongs_to_many :patients
	has_and_belongs_to_many :appointments
	has_and_belongs_to_many :doctors
	has_many :treatments
	belongs_to :medical_center #foreign key

5. Treatment
	has_and_belongs_to_many :patients
	has_and_belongs_to_many :patients
	has_and_belongs_to_many :doctors
	has_and_belongs_to_many :records
	belongs_to :medical_center #foreign key

6. MedicalCenter 
	has_many :patients
	has_many :appointments
	has_many :doctors
	has_many :records
	has_many :treatments
